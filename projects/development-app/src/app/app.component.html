Basic example

Code

component.html

<input type="text" class="form-control"
  [iconPicker]="'terminal-fill'"
  [ipPosition]="'bottom'"
  [ipWidth]="'250px'"
  [ipPlaceHolder]="'Choose an icon'"
  [ipFallbackIcon]="fallbackIconDefault"
  [value]="selectedIcon"
  (iconPickerSelect)="onIconPickerSelect($event)"
/>
